img.onload = () => { const canvas = document.createElement('canvas'); const MAX_SIZE = 1200; // Un poco más de calidad let width = img.width; let height = img.height; if (width > height) { if (width > MAX_SIZE) { height *= MAX_SIZE / width; width = MAX_SIZE; } } else { if (height > MAX_SIZE) { width *= MAX_SIZE / height; height = MAX_SIZE; } } canvas.width = width; canvas.height = height; const ctx = canvas.getContext('2d'); ctx.drawImage(img, 0, 0, width, height); canvas.toBlob((blob) => { fotoFinal = blob; // AQUÍ SE GUARDA LA FOTO PROCESADA document.getElementById('preview').src = URL.createObjectURL(blob); document.getElementById('preview-container').style.display = "block"; document.getElementById('file-text').innerText = "✅ IMAGEN LISTA"; document.getElementById('btn-procesar').disabled = false; // REHABILITAMOS }, 'image/jpeg', 0.8); }; }; }; // 🔥 PROCESO DE GUARDADO CON LINK ASEGURADO document.getElementById('btn-procesar').onclick = async () => { if(!selUF) return alert("Por favor, selecciona una unidad primero."); let linkImg = ""; // Si el usuario cargó una foto, la subimos SÍ O SÍ if(fotoFinal) { document.getElementById('loading').style.display = "block"; document.getElementById('btn-procesar').innerText = "⏳ SUBIENDO..."; linkImg = await subirCloudinary(); if(linkImg === "error" || !linkImg) { document.getElementById('loading').style.display = "none"; document.getElementById('btn-procesar').innerText = "REINTENTAR GUARDADO"; return alert("❌ No se pudo generar el link de la imagen. Intentá de nuevo."); } } else { // Si NO cargó foto, preguntamos if(!confirm("No seleccionaste ninguna foto. ¿Querés guardar la reserva igual?")) return; } try { let textoWA = ""; const dataFirebase = { fecha: window.fechaSel, uf: selUF.UF, inquilino: selUF.Inquilino, comprobanteUrl: linkImg // Aquí va el link o vacío }; if(modoReserva) { await addDoc(collection(db,"reservas_salon"), dataFirebase); textoWA = `*NUEVA RESERVA SUM*\n📍 *UF:* ${selUF.UF}\n👤 *Inquilino:* ${selUF.Inquilino}\n📅 *Fecha:* ${window.fechaSel}`; } else { await updateDoc(doc(db, "reservas_salon", window.idReserva), { comprobanteUrl: linkImg }); textoWA = `*PAGO ACTUALIZADO SUM*\n📍 *UF:* ${selUF.UF}\n👤 *Inquilino:* ${selUF.Inquilino}`; } // AGREGAR EL LINK AL MENSAJE SI EXISTE if(linkImg && linkImg !== "") { textoWA += `\n\n🔗 *Comprobante:* ${linkImg}`; } const btnWA = document.getElementById('btn-wa'); btnWA.href = `https://wa.me/${PHONE}?text=${encodeURIComponent(textoWA)}`; // UI FINAL document.getElementById('btn-procesar').style.display = "none"; document.getElementById('loading').style.display = "none"; btnWA.style.display = "flex"; alert("✅ Guardado con éxito.
